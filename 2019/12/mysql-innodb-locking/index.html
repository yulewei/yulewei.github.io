排他锁

session1 session2 
mysql> set autocommit = 0;
Query OK, 0 rows affected (0.00 sec)

-- 在 id = 3 的记录上添加排他锁

mysql> update tbl set age = 100 where id = 3;

Query OK, 1 row affected (0.00 sec)

Rows matched: 1  Changed: 1  Warnings: 0


mysql> set autocommit = 0;
Query OK, 0 rows affected (0.00 sec)

-- 锁等待超时时间修改为 5 秒

mysql> set @@innodb_lock_wait_timeout = 5;

Query OK, 0 rows affected (0.00 sec)


-- 其他 session 可以查询 id = 3 的记录，但是不能获取该记录的排他锁

mysql> select * from tbl where id = 3;

+----+------+------+

| id | name | age  |

+----+------+------+

|  3 | Tom3 |   23 |

+----+------+------+

1 row in set (0.00 sec)


-- 等待 5 秒

mysql> update tbl set age = 100 where id = 3;

ERROR 1205 (HY000): Lock wait timeout exceeded; try restarting transaction


-- 可以获得其他记录的排他锁

mysql> update tbl set age = 100 where id = 2;

Query OK, 1 row affected (0.00 sec)

Rows matched: 1  Changed: 1  Warnings: 0


-- 等待 5 秒

mysql> update tbl set age = 100 where id >= 2;

ERROR 1205 (HY000): Lock wait timeout exceeded; try restarting transaction


mysql> rollback;
Query OK, 0 rows affected (0.00 sec)
 
mysql> rollback;
Query OK, 0 rows affected (0.00 sec)

|  3 | Tom3 |   23 |

-- 等待

mysql> select * from tbl where id = 3 for update;

ERROR 1205 (HY000): Lock wait timeout exceeded; try restarting transaction


-- 可以获得其他记录的排他锁

mysql> select * from tbl where id = 2 for update;

+----+------+------+

| id | name | age  |

+----+------+------+

|  2 | Tom3 |   22 |

+----+------+------+

1 row in set (0.00 sec)


-- 等待

mysql> delete from tbl where id = 3;

ERROR 1205 (HY000): Lock wait timeout exceeded; try restarting transaction


-- 等待

mysql> delete from tbl where id >= 2;

ERROR 1205 (HY000): Lock wait timeout exceeded; try restarting transaction


mysql> rollback;

mysql> rollback;
Query OK, 0 rows affected (0.00 sec)

记录不存在问题

insert into tbl values (5, 'Tom5', 25);
insert into tbl values (6, 'Tom6', 26);
insert into tbl values (1, 'Tom1', 21);

会话1
mysql> set autocommit = 0;
Query OK, 0 rows affected (0.00 sec)

mysql> select * from tbl where id > 1 for update;
+----+------+------+
| id | name | age |
+----+------+------+
| 2 | Tom2 | 22 |
| 3 | Tom3 | 23 |
| 4 | Tom4 | 24 |
+----+------+------+
3 rows in set (0.00 sec)

mysql> set autocommit = 0;
Query OK, 0 rows affected (0.00 sec)

mysql> select * from tbl where id = 5 for update;
Empty set (0.00 sec)

会话2
mysql> set autocommit = 0;
Query OK, 0 rows affected (0.00 sec)

mysql> select * from tbl where id = 5 for update;
Empty set (0.00 sec)

mysql> insert into tbl values (1, 'Tom1', 21);
Query OK, 1 row affected (0.00 sec)

mysql> insert into tbl values (5, 'Tom5', 25);
ERROR 1205 (HY000): Lock wait timeout exceeded; try restarting transaction

mysql> insert into tbl values (6, 'Tom6', 26);
ERROR 1205 (HY000): Lock wait timeout exceeded; try restarting transaction
